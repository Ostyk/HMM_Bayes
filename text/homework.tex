\subsection{Expression for TopOfTheHour}
\begin{equation}
\begin{aligned}
P(T \mid W, C, A) &= \alpha P (W, C, A \mid T) P(T) \\
&= \alpha P (W \mid T) P (C \mid T) P (A \mid T) P(T) \\
&= \alpha P(W) P(C \mid T) P (A) P(T) \\
&= P(T \mid C) P(W) P (A) 
\end{aligned}
\end{equation}
